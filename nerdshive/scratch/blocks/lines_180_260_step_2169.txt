  const handleDownloadCSV = async () => {
    try {
      const { apiClient } = await import('@/lib/apiClient');
      const response = await apiClient.get('/users/download-template?filename=employee_enrollment_template.csv', {
        responseType: 'blob'
      });
      const blob = new Blob([response.data], { type: "text/csv;charset=utf-8;" });
      const url = URL.createObjectURL(blob);
      const link = document.createElement("a");
      link.href = url;
      link.download = "employee_enrollment_template.csv";
      document.body.appendChild(link);
      link.click();
      document.body.removeChild(link);
      setTimeout(() => {
        URL.revokeObjectURL(url);
      }, 100);

      toast({
        title: "Template Downloaded",
        description: "Standardized employee template has been downloaded.",
      });
    } catch (error) {
      console.error("Error downloading template", error);

        title: "Download Failed",

      });
    }
  };
